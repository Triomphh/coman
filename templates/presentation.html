Configuration de conteneurisation pour un déploiement cohérent et reproductible.

  • • Image basée sur GCC
  • • Installation automatisée des dépendances
  • • Compilation intégrée 
FROM gcc:latest

WORKDIR /app
COPY . .

RUN apt-get update && apt-get install -y \
    cmake \
    build-essential \
    libasio-dev

RUN mkdir build && cd build && \
    cmake .. && \
    make

EXPOSE 18080

CMD ["./coman"]

Tests Unitaires

Tests automatisés avec Google Test Framework pour garantir la fiabilité du code.

  • • Framework Google Test
  • • Tests unitaires exhaustifs
  • • Assertions avancées
  • • Intégration CI/CD 
#include <gtest/gtest.h>
#include "Project.h"

TEST(ProjectTest, Creation) {
    Project project("Test Project");
    EXPECT_EQ(project.getName(), "Test Project");
    EXPECT_TRUE(project.getTasks().empty());
}

TEST(ProjectTest, AddTask) {
    Project project("Test Project");
    Task task("Test Task");
    
    project.addTask(task);
    EXPECT_EQ(project.getTasks().size(), 1);
    EXPECT_EQ(project.getTasks()[0].getName(), "Test Task");
}

TEST(ProjectTest, TaskStatus) {
    Project project("Test Project");
    Task task("Test Task");
    
    project.addTask(task);
    EXPECT_EQ(project.getTasks()[0].getStatus(), TaskStatus::TODO);
    
    project.getTasks()[0].setStatus(TaskStatus::IN_PROGRESS);
    EXPECT_EQ(project.getTasks()[0].getStatus(), TaskStatus::IN_PROGRESS);
}
